2.  must include below code before using passport in index.js
    app.use(
    session({
        secret: "myscript",     //Used to sign the session ID cookie so it can’t be tampered with.
                                //Should be a long, random string in production, not "myscript".
        resave: false,          //If false → session isn’t saved back to the store if it hasn’t been modified during the request.
                                //This reduces unnecessary writes (especially if you store sessions in a database like PostgreSQL later).
                                //If true → every request saves the session again, even if nothing changed (not recommended unless your store requires it).
        saveUninitialized: true, // If true → store a session even if it’s new and has no data.
                                //Useful for tracking visitors before they log in (e.g., analytics, CSRF tokens).
                                //If false → don’t save empty sessions; only save after something is set (like req.session.user).
                                //In authentication scenarios, false is usually better for performance and privacy.
    })
    );

    router.post("/", (req, res, next) => {
      passport.authenticate("local", (err, user, info) => {
        if (err) {
          return next(err);
        }
        if (!user) {
          return res.render("index.ejs", {
            error: info?.message || "Login failed",
            username: req.body.username,
          });
        }
        req.logIn(user, (err) => {
          if (err) {
            return next(err);
          }
          return res.redirect("/dashboard");
        });
      })(req, res, next);
    });
